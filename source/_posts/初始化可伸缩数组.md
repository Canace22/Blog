## Demo

```js
let len = 5;

// method1
let arr = Array(len)
  .fill(null)
  .map((v, i) => String.fromCharCode(65 + i));
// method2
let arr = Array.apply(null, { length: len }).map((v, i) => String.fromCharCode(65 + i));

console.log(arr); //[ 'A', 'B', 'C', 'D', 'E' ]
```
